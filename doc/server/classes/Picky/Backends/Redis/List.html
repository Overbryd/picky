def []= key, values
  delete key

  redis_key = "#{namespace}:#{key}"
  i = 0
  values.each do |value|
    i += 1
    client.zadd redis_key, i, value
  end

  DirectlyManipulable.make self, values, key
  values
end

def size
  redis_key = "#{namespace}:*"
  client.keys(redis_key).inject(0) do |total, key|
    total + client.zcard(key)
  end
end
