def hashify list
  list.inject({}) do |total, element|
    if code = encoded(element)
      total[code] ||= []
      total[code] << element
    end
    total
  end
end

def sort hash
  hash.each_pair.each do |code, ary|
    ary.sort_by_levenshtein! code
    ary.slice! amount, ary.size # size is not perfectly correct, but anyway
  end
  hash
