def initialize from, to
  @from, @to = from, to

  if @to.zero?
    def each_subtoken token, &block
      token.each_subtoken @from, &block
    end
  else
    if @from < 0 && @to < 0
      def each_subtoken token, &block
        token.each_subtoken @from - @to - 1, (0..@to), &block
      end
    else
      def each_subtoken token, &block
        token.each_subtoken @from, (0..@to), &block
      end
    end
  end
